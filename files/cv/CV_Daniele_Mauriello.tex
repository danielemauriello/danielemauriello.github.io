%----------------------------------------------------------------------------------------
%	\cventry: renders one CV entry (education / experience item) as a two-column row.
%
%	  Left column  (~24% of line width) - date and place, stacked, italicized.
%	  Right column (~73% of line width) - institution / role / description, stacked.
%
%	Arguments:
%	  #1 - Left column content. Stack multiple lines with "\\" (e.g. period \\ place).
%	  #2 - Right column content. Stack multiple lines with "\\"
%	       (e.g. institution in \textbf{} \\ role \\ description).
%
%	Usage:
%	  \cventry{Oct 2021 - Ongoing \\ Bonn, Germany}%
%	          {\textbf{Bonn Graduate School of Economics} \\ Ph.D. in Economics}
%
%	Note: relies on \ifthenelse from the "ifthen" package, already loaded by resume.cls.
%----------------------------------------------------------------------------------------
\newcommand{\cventry}[2]{%
  \noindent
  \begin{minipage}[t]{0.24\linewidth}
    \raggedright\textcolor{gray}{\emph{#1}}
  \end{minipage}%
  \hspace{0.02\linewidth}%
  \begin{minipage}[t]{0.73\linewidth}
    \raggedright #2
  \end{minipage}
  \par\medskip
}
\newcommand{\cvtitle}[1]{\textbf{#1}}
\newcommand{\cvrole}[1]{\emph{#1}}
\newcommand{\cvdesc}[1]{{\small\color{gray}#1}}
\newcommand{\cvaward}[2]{%
  \noindent#1\hfill\textcolor{gray}{\small #2}\par\smallskip
}

%----------------------------------------------------------------------------------------
%	CUSTOM HEADER STYLE (overrides resume.cls's default all-caps \printname)
%	Centered small-caps name, with contact details on a single line below it,
%	separated by "|". No rule beneath the header.
%	Redefines \printname, which resume.cls calls automatically at \begin{document}
%	via \@name (set by \name{}) and \@addresstwo (set by \address{}).
%----------------------------------------------------------------------------------------
\makeatletter
\def\printname{
  \begingroup
    \centering
    {\fontsize{20}{22}\selectfont\scshape \@name}\par
    \smallskip
    {\small \@addresstwo}\par
    \bigskip
  \endgroup
}
\def\printaddress#1{} % Suppress the default centered address line (handled above instead)
\makeatother
